int []a=new int[47];//JAVA陣列
void setup(){
  size(500,200);//大一點
  textSize(30);
  for(int i=0;i<47;i++)a[i]=i;
//讓a[i]的陣列裡面要先放整齊對應的數字
  for(int i=0;i<1000;i++){
    int i1 = (int)random(47);
    int i2 = (int)random(47);
    int temp=a[i1];a[i1]=a[i2];a[i2]=temp;
  }//作弊 先洗好牌 先知道得獎號碼 等下再掉出來
}
int N=0;
void draw(){
  background(#62AED3);
  for(int i=0;i<N;i++){
    text(a[i],i*80,100);
  }
}
void mousePressed(){
  N++;
}
